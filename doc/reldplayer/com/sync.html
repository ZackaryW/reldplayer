138    def _wrapOver(self, func, callingKey: str = "mnq_name"):
139        """
140        takes a function and wraps it so that it can repeatly be called
141        """
142        if self.__currentSyncKey is None:
143            return func
144        # check mnq_name in signature
145        if callingKey not in inspect.signature(func).parameters:
146            return func
147
148        if func.__name__ in ["copy", "add"]:
149            raise RuntimeError("copy and add are not supported")
150
151        def wrapper(*args, **kwargs):
152            assert self.__currentSyncKey in self.__dictionary, "Key doesn't exist"
153            for id in self.get(self.__currentSyncKey):
154                func(*args, **kwargs, mnq_name=id)
155                if self.__get_config__.SYNC_DELAY > 0:
156                    sleep(self.__get_config__.SYNC_DELAY)
157
158        return wrapper
