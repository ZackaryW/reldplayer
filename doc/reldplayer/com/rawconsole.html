 58    def _queryRaw(self, *args, timeout: int = 5):
 59        """
 60        Executes a raw query using the specified arguments and a timeout.
 61
 62        Args:
 63            *args: Any additional arguments for the query.
 64            timeout (int): The maximum time to wait for the query to complete (default is 5).
 65
 66        Returns:
 67            bytes: The output of the query execution.
 68        """
 69        try:
 70            if not len(args):
 71                queryed = [self.__ldconsole__]
 72            else:
 73                queryed = [self.__ldconsole__, *(str(arg) for arg in args)]
 74
 75            proc: subprocess.CompletedProcess = subprocess.run(
 76                queryed, capture_output=True, timeout=timeout
 77            )
 78            comm: bytes = proc.stdout
 79
 80        except subprocess.TimeoutExpired as e:
 81            raise e
 82        except subprocess.CalledProcessError as e:
 83            raise e
 84
 85        return comm

 87    def _query(
 88        self,
 89        *args,
 90        timeout: int = 5,
 91        raw: bool = False,
 92        decodeOrder: typing.List[str] = ["utf-8", "gbk"],
 93        toList: bool = False,
 94        stripNullLines: bool = False,
 95    ):
 96        """
 97        Perform a query with optional parameters for timeout, raw output, return context, decoding order, conversion to list, and stripping null lines.
 98
 99        Args:
100            *args: Variable length argument list.
101            timeout (int): Timeout duration in seconds (default is 5).
102            raw (bool): Flag to indicate whether to return raw output (default is False).
103            decodeOrder (List[str]): List of encoding orders to decode raw output (default is ["utf-8", "gbk"]).
104            toList (bool): Flag to indicate whether to convert the output to a list (default is False).
105            stripNullLines (bool): Flag to indicate whether to strip null lines from the output (default is False).
106
107        Returns:
108            str: The processed query output
109        """
110        rawProcess = self._queryRaw(*args, timeout=timeout)
111
112        if raw:
113            return rawProcess
114
115        for decoder in decodeOrder:
116            try:
117                decoded = rawProcess.decode(decoder)
118            except:  # noqa
119                continue
120
121            ret = decoded
122
123        if toList:
124            ret = ret.strip().split("\r\n")
125
126            if stripNullLines:
127                ret = list(filter(None, ret))
128                ret = list(map(lambda x: x.strip(), ret))
129
130        return ret

 79            comm: bytes = proc.stdout
 80
 81        except subprocess.TimeoutExpired as e:
 82            raise e
 83        except subprocess.CalledProcessError as e:
 84            raise e
 85
 86        return comm

 88    def _query(
 89        self,
 90        *args,
 91        timeout: int = 5,
 92        raw: bool = False,
 93        decodeOrder: typing.List[str] = ["utf-8", "gbk"],
 94        toList: bool = False,
 95        stripNullLines: bool = False,
 96    ):
 97        """
 98        Perform a query with optional parameters for timeout, raw output, return context, decoding order, conversion to list, and stripping null lines.
 99
100        Args:
101            *args: Variable length argument list.
102            timeout (int): Timeout duration in seconds (default is 5).
103            raw (bool): Flag to indicate whether to return raw output (default is False).
104            decodeOrder (List[str]): List of encoding orders to decode raw output (default is ["utf-8", "gbk"]).
105            toList (bool): Flag to indicate whether to convert the output to a list (default is False).
106            stripNullLines (bool): Flag to indicate whether to strip null lines from the output (default is False).
107
108        Returns:
109            str: The processed query output
110        """
111        rawProcess = self._queryRaw(*args, timeout=timeout)
112
113        if raw:
114            return rawProcess
115
116        for decoder in decodeOrder:
117            try:
118                decoded = rawProcess.decode(decoder)
119            except:  # noqa
120                continue
121
122            ret = decoded
123
124        if toList:
125            ret = ret.strip().split("\r\n")
126
127            if stripNullLines:
128                ret = list(filter(None, ret))
129                ret = list(map(lambda x: x.strip(), ret))
130
131        return ret
